cmake_minimum_required(VERSION 3.5)

project(UnityConnection)

#-----------------------------------------------------------------------------
# Extension meta-information
set(EXTENSION_HOMEPAGE "http://slicer.org/slicerWiki/index.php/Documentation/Nightly/Extensions/UnityConnection")
set(EXTENSION_CATEGORY "Examples")
set(EXTENSION_CONTRIBUTORS "Thomas Muender (University Bremen)")
set(EXTENSION_DESCRIPTION "Connection to Unity3D")
set(EXTENSION_ICONURL "http://www.example.com/Slicer/Extensions/UnityConnection.png")
set(EXTENSION_SCREENSHOTURLS "http://www.example.com/Slicer/Extensions/UnityConnection/Screenshots/1.png")
set(EXTENSION_DEPENDS "NA") # Specified as a space separated string, a list or 'NA' if any

#-----------------------------------------------------------------------------
# Extension dependencies
find_package(Slicer REQUIRED)
include(${Slicer_USE_FILE})

#-----------------------------------------------------------------------------
# Extension modules
add_subdirectory(Connector)
## NEXT_MODULE

#-----------------------------------------------------------------------------
include(${Slicer_EXTENSION_GENERATE_CONFIG})
include(${Slicer_EXTENSION_CPACK})
